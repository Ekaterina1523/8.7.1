<body>

    <h1>Компьютер угадывает ваше число!</h1>

    <div id="gameArea">

        <label for="min">Минимум:</label>

        <input type="number" id="min" value="0">

        <br>

        <label for="max">Максимум:</label>

        <input type="number" id="max" value="100">

        <br>

        <button id="startButton">Начать игру</button>

        <button id="resetButton" style="display:none;">Заново</button>

        <p id="question"></p>

    </div>

    <div id="responseArea" style="display:none;">

        <p>Компьютер думает, что это число: <span id="currentGuess"></span></p>

        <button id="higherButton">Больше</button>

        <button id="lowerButton">Меньше</button>

        <button id="correctButton">Правильно!</button>

    </div>

    <p id="message"></p>


    <script>

        let min, max, guess, attempts;


        function startGame() {

            min = parseInt(document.getElementById('min').value);

            max = parseInt(document.getElementById('max').value);

            attempts = 0;

            document.getElementById('message').innerText = '';

            document.getElementById('gameArea').style.display = 'none';

            document.getElementById('responseArea').style.display = 'block';

            makeGuess();

        }


        function makeGuess() {

            guess = Math.round((min + max) / 2);

            document.getElementById('currentGuess').innerText = numberToText(guess);

            attempts++;

            document.getElementById('question').innerText = `Попытка ${attempts}: ` + guess;

        }


        function numberToText(num) {

            if (num === 0) return "ноль";

            let isNegative = num < 0;

            if (isNegative) num = Math.abs(num);


            let parts = [];

            if (Math.floor(num / 100) > 0) {

                parts.push(['сто', 'двести', 'триста', 'четыреста', 'пятьсот', 'шестьсот', 'семьсот', 'восемьсот', 'девятьсот'][Math.floor(num / 100)]);

                num = num % 100;

            }

            if (num >= 20) {

                parts.push(['двадцать', 'тридцать', 'сорок', 'пятьдесят', 'шестьдесят', 'семьдесят', 'восемьдесят', 'девяносто'][Math.floor(num / 10) - 2]);

                num = num % 10;

            }

            if (num >= 1) {

                parts.push(['один', 'два', 'три', 'четыре', 'пять', 'шесть', 'семь', 'восемь', 'девять'][num - 1]);

            }


            let result = parts.join(' ');

            return isNegative ? 'минус ' + result : result;

        }


        document.getElementById('startButton').onclick = startGame;


        document.getElementById('higherButton').onclick = function() {

            min = guess + 1;

            makeGuess();

        };


        document.getElementById('lowerButton').onclick = function() {

            max = guess - 1;

            makeGuess();

        };


        document.getElementById('correctButton').onclick = function() {

            document.getElementById('message').innerText = `Компьютер угадал ваше число ${numberToText(guess)} за ${attempts} попыток!`;

            document.getElementById('resetButton').style.display = 'inline';

            document.getElementById('responseArea').style.display = 'none';

        };


        document.getElementById('resetButton').onclick = function() {

            document.getElementById('gameArea').style.display = 'block';

            document.getElementById('resetButton').style.display = 'none';

            document.getElementById('responseArea').style.display = 'none';

        };

    </script>
